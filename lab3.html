	    <!ATTLIST project projid ID #REQUIRED >
		<!ELEMENT title (#PCDATA) >
		<!ELEMENT managers (manager+) >
			<!ELEMENT manager (name, mphone, memail*) >
				<!ELEMENT name (firstname, middlename?, lastname) >
					<!ELEMENT firstname (#PCDATA) >
					<!ELEMENT middlename (#PCDATA) >
					<!ELEMENT lastname (#PCDATA) >
				<!ELEMENT mphone EMPTY >	
				<!ATTLIST mphone office CDATA #REQUIRED >
				<!ATTLIST mphone home CDATA #REQUIRED >	
				<!ELEMENT memail (#PCDATA) >			
		<!ELEMENT teams (team+) >		
			<!ELEMENT team (leader, developers+, tester?, documenters*) >
				<!ELEMENT leader (#PCDATA) >
				<!ELEMENT developers (#PCDATA) >
				<!ELEMENT tester (#PCDATA) >	
				<!ELEMENT documenters (#PCDATA) >
		<!ELEMENT clients (client?) >
			<!ELEMENT client (clientname, address, clientemail, clientphone) >
			<!ATTLIST client clientid ID #REQUIRED>
				<!ELEMENT clientname (#PCDATA) >
